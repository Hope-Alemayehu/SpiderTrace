\begin{tikzpicture}[
  node distance = 0.55cm and 1.1cm,
  box/.style     = {rectangle, rounded corners=3pt, minimum width=2.2cm,
                    minimum height=0.65cm, align=center, font=\small},
  inputbox/.style = {box, draw=black!60, fill=blue!8},
  hiddenbox/.style = {box, draw=black!60, fill=gray!12},
  headbox/.style   = {box, draw=black!60, minimum width=2.0cm},
  outputbox/.style = {box, draw=black!60, minimum width=1.6cm,

  arrow/.style   = {-{Stealth[length=5pt]}, thick},

]

%% ── Input ────────────────────────────────────────────────────────────────
\node[inputbox] (syndrome)
  {Syndrome bits\\$\mathbf{s} \in \{0,1\}^{n_s}$};

%% ── Backbone ─────────────────────────────────────────────────────────────
\node[hiddenbox, right=1.3cm of syndrome] (h1)
  {Linear(256)\\ReLU, Dropout};

\node[hiddenbox, right=0.7cm of h1] (h2)
  {Linear(128)\\ReLU, Dropout};

%% ── Shared representation ────────────────────────────────────────────────
\node[hiddenbox, right=0.7cm of h2, fill=amber!15, draw=amber!60!black]
  (h) {$\mathbf{h}$ (128-dim)};

\definecolor{amber}{RGB}{186,117,23}

%% ── Flip head ────────────────────────────────────────────────────────────
\node[headbox, above right=0.9cm and 1.1cm of h,
      fill=green!10, draw=green!50!black] (fliphead)
  {Flip head\\Linear(1) + $\sigma$};

%% ── ZX head ──────────────────────────────────────────────────────────────
\node[headbox, below right=0.9cm and 1.1cm of h,
      fill=red!8, draw=red!40!black] (zxhead)
  {ZX head\\Linear($4n$)};

%% ── Outputs ──────────────────────────────────────────────────────────────
\node[outputbox, right=0.9cm of fliphead,
      fill=green!15, draw=green!50!black] (flipout)
  {$\hat{y} \in [0,1]$};

\node[outputbox, right=0.9cm of zxhead,
      fill=red!10, draw=red!40!black] (zxout)
  {$\hat{\mathbf{f}} \in \mathbb{R}^{4n}$};

%% ── Loss labels ──────────────────────────────────────────────────────────
\node[label, right=0.3cm of flipout, text=green!40!black]
  (bcelabel) {BCE loss\\$\mathcal{L}_\mathrm{flip}$};

\node[label, right=0.3cm of zxout, text=red!50!black]
  (celabel) {CE loss\\$\mathcal{L}_\mathrm{zx}$};

%% ── Arrows: input → backbone → h ─────────────────────────────────────────
\draw[arrow] (syndrome) -- (h1);
\draw[arrow] (h1) -- (h2);
\draw[arrow] (h2) -- (h);

%% ── Arrows: h → heads ────────────────────────────────────────────────────
\draw[arrow] (h.north east) -- (fliphead.west);
\draw[arrow] (h.south east) -- (zxhead.west);

%% ── Arrows: heads → outputs ──────────────────────────────────────────────
\draw[arrow] (fliphead) -- (flipout);
\draw[arrow] (zxhead) -- (zxout);

%% ── Arrows: outputs → losses ─────────────────────────────────────────────
\draw[arrow] (flipout) -- (bcelabel);
\draw[arrow] (zxout)   -- (celabel);

%% ── Inference annotation: ZX head crossed out ────────────────────────────
\node[label, below=0.18cm of zxhead, text=red!60!black, font=\scriptsize]

  (zxhead.north west) -- (zxhead.south east);

  (zxout.north west) -- (zxout.south east);

\node[label, above=0.5cm of h1, xshift=0.5cm, text=gray!70!black,
